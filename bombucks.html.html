  <script>
    const texts = {
      ru: { header: '@GPT_gamer_bot', signal: 'Получить сигнал' },
      en: { header: '@GPT_gamer_bot', signal: 'Get Signal' }
    };

    function selectLang(lang) {
      document.getElementById('language-modal').style.display = 'none';
      document.getElementById('overlay').style.display = 'none';
      document.getElementById('header-text').textContent = texts[lang].header;
      document.getElementById('get-signal').textContent = texts[lang].signal;
    }

    const sizes = [
      [2, 3], [3, 6], [4, 9], [5, 12], [6, 15]
    ];
    let currentSizeIndex = 0;
    const sizeDisplay = document.getElementById('size-display');
    const gridEl = document.getElementById('grid');

    function renderGrid() {
      const [cols, rows] = sizes[currentSizeIndex];
      sizeDisplay.textContent = `${cols}×${rows}`;
      gridEl.style.gridTemplateColumns = `repeat(${cols}, var(--cell-size))`;
      gridEl.innerHTML = '';
      for (let i = 0; i < cols * rows; i++) {
        const cell = document.createElement('div');
        cell.className = 'cell';
        gridEl.appendChild(cell);
      }
    }

    document.getElementById('prev-size').onclick = () => {
      if (currentSizeIndex > 0) currentSizeIndex--;
      renderGrid();
    };
    document.getElementById('next-size').onclick = () => {
      if (currentSizeIndex < sizes.length - 1) currentSizeIndex++;
      renderGrid();
    };

    function generatePath() {
      const cells = Array.from(gridEl.children);
      const [cols, rows] = sizes[currentSizeIndex];
      cells.forEach(c => {
        c.textContent = '';
        c.classList.remove('flash');
      });

      const path = [];
      for (let row = rows - 1; row >= 0; row--) {
        const start = row * cols;
        path.push(start + Math.floor(Math.random() * cols));
      }

      path.forEach((idx, step) => {
        setTimeout(() => {
          const cell = cells[idx];
          cell.classList.add('flash');
          setTimeout(() => { cell.textContent = '★'; }, 150);
          setTimeout(() => { cell.classList.remove('flash'); }, 1000);
        }, step * 300);
      });
    }

    document.getElementById('get-signal').onclick = generatePath;

    renderGrid();
  </script>

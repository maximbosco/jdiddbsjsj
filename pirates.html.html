  <script>
    const langStrings = {
      ru: {
        title: "@GPT_gamer_bot",
        choose_skull: "Выберите череп",
        shuffle: "Перемешать",
        again: "Ещё раз"
      },
      en: {
        title: "@GPT_gamer_bot",
        choose_skull: "Choose a skull",
        shuffle: "Shuffle",
        again: "Play again"
      }
    };

    let currentLang = 'ru';
    const shells = document.querySelectorAll('.shell');
    const button = document.getElementById('actionButton');
    let selectedShell = null;
    let phase = 'choose';

    function setLanguage(lang) {
      currentLang = lang;
      document.querySelector('.lang-modal').style.display = 'none';
      updateTexts();
      resetGame();
    }

    function updateTexts() {
      document.querySelectorAll('[data-lang]').forEach(el => {
        const key = el.dataset.lang;
        el.textContent = langStrings[currentLang][key];
      });
    }

    shells.forEach(shell => {
      shell.addEventListener('click', () => {
        if (phase !== 'choose') return;
        shells.forEach(s => s.classList.remove('selected'));
        shell.classList.add('selected');
        selectedShell = parseInt(shell.dataset.index);
        button.textContent = langStrings[currentLang].shuffle;
        button.classList.add('active');
        button.disabled = false;
      });
    });

    button.addEventListener('click', () => {
      if (phase === 'choose' && selectedShell !== null) {
        phase = 'shuffle';
        button.classList.remove('active');
        button.disabled = true;

        shells.forEach(shell => {
          shell.classList.remove('selected');
          shell.classList.add('shaking'); // Add shake animation
        });

        shuffleShells(() => {
          revealCrystal();
          button.textContent = langStrings[currentLang].again;
          button.classList.add('active');
          button.disabled = false;
          phase = 'done';
          shells.forEach(shell => shell.classList.remove('shaking'));
        });

      } else if (phase === 'done') {
        resetGame();
      }
    });

    function shuffleShells(callback) {
      const container = document.getElementById('shells');
      let count = 0;

      const interval = setInterval(() => {
        const shuffled = Array.from(shells).sort(() => Math.random() - 0.5);
        shuffled.forEach((el) => container.appendChild(el));
        count++;
        if (count >= 20) {
          clearInterval(interval);
          callback();
        }
      }, 100);
    }

    function revealCrystal() {
      const randomIndex = Math.floor(Math.random() * 3);
      shells.forEach(s => s.classList.remove('highlight'));
      shells[randomIndex].classList.add('highlight');
    }

    function resetGame() {
      shells.forEach(shell => {
        shell.classList.remove('selected', 'highlight');
      });
      selectedShell = null;
      phase = 'choose';
      button.textContent = langStrings[currentLang].choose_skull;
      button.classList.remove('active');
      button.disabled = true;
    }

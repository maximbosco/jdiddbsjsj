  <script>
    let minesCount = 3;
    let currentMinesIndex = 1;
    const availableMines = [1, 3, 5, 7];
    const minesCountElement = document.getElementById('mines-count');
    const gameGrid = document.getElementById('game-grid');
    const cells = [];
    const translations = {
      ru: {
        header: 'MINES',
        showStars: 'ПОЛУЧИТЬ СИГНАЛ',
        chooseLanguage: 'Выберите язык',
        minesCount: '3'
      },
      en: {
        header: 'MINES',
        showStars: 'GET SIGNAL',
        chooseLanguage: 'Choose language',
        minesCount: '3'
      }
    };

    function selectLang(lang) {
      document.getElementById('language-modal').style.display = 'none';
      document.getElementById('header-text').textContent = translations[lang].header;
      document.getElementById('get-signal').textContent = translations[lang].showStars;
    }

    function initGameGrid() {
      gameGrid.innerHTML = '';
      cells.length = 0;
      for (let i = 0; i < 25; i++) {
        const cell = document.createElement('div');
        cell.className = 'cell';
        cells.push(cell);
        gameGrid.appendChild(cell);
      }
    }

    function updateMinesCount() {
      minesCount = availableMines[currentMinesIndex];
      minesCountElement.textContent = minesCount;
    }

    function generateSignal() {
      cells.forEach(cell => {
        cell.textContent = '';
        cell.classList.remove('revealed');
      });
      const mineIndices = new Set();
      while (mineIndices.size < minesCount) {
        mineIndices.add(Math.floor(Math.random() * 25));
      }
      const safeCells = Array.from({ length: 25 }, (_, i) => i).filter(i => !mineIndices.has(i));
      const revealedSafe = safeCells.sort(() => Math.random() - 0.5).slice(0, 5);
      revealedSafe.forEach((i, index) => {
        setTimeout(() => {
          const cell = cells[i];
          cell.textContent = '★';
          cell.style.color = '#FFD60A';
          cell.classList.add('revealed');
        }, index * 200);
      });
    }

    document.getElementById('decrease-mines').addEventListener('click', () => {
      if (currentMinesIndex > 0) {
        currentMinesIndex--;
        updateMinesCount();
      }
    });

    document.getElementById('increase-mines').addEventListener('click', () => {
      if (currentMinesIndex < availableMines.length - 1) {
        currentMinesIndex++;
        updateMinesCount();
      }
    });

    document.getElementById('get-signal').addEventListener('click', generateSignal);

    initGameGrid();
    updateMinesCount();
  </script>

  <script>
    let currentLanguage = 'ru';
    let lastResult = '';
    let streakCount = 0;
    const translations = {
      ru: {
        header: '@GPT_gamer_bot',
        flipCoin: 'Подбросить монетку',
        chooseLanguage: 'Выберите язык',
        result: 'Результат:'
      },
      en: {
        header: '@GPT_gamer_bot',
        flipCoin: 'Flip the coin',
        chooseLanguage: 'Choose language',
        result: 'Result:'
      },
      results: {
        ru: ['Орел', 'Решка'],
        en: ['Heads', 'Tails']
      }
    };

    function selectLang(lang) {
      currentLanguage = lang;
      document.getElementById('language-modal').style.display = 'none';
      document.getElementById('header-text').textContent = translations[lang].header;
      document.getElementById('flip-button').textContent = translations[lang].flipCoin;
      document.getElementById('result').textContent = translations[lang].result;
    }

    function updateResult(result) {
      if (result === lastResult) {
        streakCount++;
        document.getElementById('result').textContent = `${translations[currentLanguage].result} ${result} x${streakCount}`;
      } else {
        lastResult = result;
        streakCount = 1;
        document.getElementById('result').textContent = `${translations[currentLanguage].result} ${result}`;
      }
    }

    document.getElementById('flip-button').addEventListener('click', () => {
      const coin = document.getElementById('coin');
      const resultText = document.getElementById('result');
      coin.classList.add('flip');
      
      setTimeout(() => {
        coin.classList.remove('flip');
        const result = Math.random() < 0.5 ? translations.results[currentLanguage][0] : translations.results[currentLanguage][1];
        updateResult(result);
      }, 1000);
    });
  </script>

  <script>
    const goalGrid = document.getElementById('goal-grid');
    const cells = [];
    const translations = {
      ru: {
        header: '@GPT_gamer_bot',
        getSignal: 'ПОЛУЧИТЬ СИГНАЛ',
        chooseLanguage: 'Выберите язык',
        title: 'PENALTY'
      },
      en: {
        header: '@GPT_gamer_bot',
        getSignal: 'GET SIGNAL',
        chooseLanguage: 'Choose language',
        title: 'PENALTY'
      }
    };
    function selectLang(lang) {
      document.getElementById('language-modal').style.display = 'none';
      document.getElementById('header-text').textContent = translations[lang].header;
      document.getElementById('get-signal').textContent = translations[lang].getSignal;
      document.getElementById('title-text').textContent = translations[lang].title;
    }
    function initGoalGrid() {
      goalGrid.innerHTML = '';
      cells.length = 0;
      for (let i = 0; i < 15; i++) {
        const cell = document.createElement('div');
        cell.className = 'cell';
        cells.push(cell);
        goalGrid.appendChild(cell);
      }
    }
    function revealShot() {
      cells.forEach(cell => cell.innerHTML = '');
      const index = Math.floor(Math.random() * cells.length);
      const glow = document.createElement('div');
      glow.className = 'ball-glow';
      const ball = document.createElement('span');
      ball.textContent = '⚽️';
      ball.className = 'ball';
      cells[index].appendChild(glow);
      cells[index].appendChild(ball);
    }
    document.getElementById('get-signal').addEventListener('click', revealShot);
    initGoalGrid();
  </script>
